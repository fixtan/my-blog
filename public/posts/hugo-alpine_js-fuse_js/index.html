        threshold: 0.3,
      });
    },

    search() {
      if (this.fuse && this.searchQuery.trim() !== "") {
        const searchResults = this.fuse.search(this.searchQuery);
        this.results = searchResults.map(r => r.item);
      } else {
        this.results = [];
      }
    },
  };
}
